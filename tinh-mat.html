function getSpellingGuide(word) { // 1. Danh sách tất cả các vần phức tạp (3-4 chữ cái) chuẩn sách giáo khoa const danhSachVanPhucTap = { "oay": "o, a, i, oay", "oai": "o, a, i, oai", "oan": "o, a, nờ, oan", "oang": "o, a, ngờ, oang", "oanh": "o, a, nhờ, oanh", "uynh": "u, i, nhờ, uynh", "uich": "u, i, chờ, uịch", "uyêt": "u, iê, tờ, uyết", "uyên": "u, iê, nờ, uyên", "ươu": "ươ, u, ươu", "ương": "ươ, ngờ, ương", "ươn": "ươ, nờ, ươn", "iêu": "i, ê, u, iêu", "yêu": "i, ê, u, yêu", "iên": "i, ê, nờ, iên", "yên": "i, ê, nờ, yên", "ang": "a, ngờ, ang", "anh": "a, nhờ, anh", "inh": "i, nhờ, inh", "ich": "i, chờ, ích", "ăng": "ă, ngờ, ăng", "âng": "â, ngờ, âng", "ong": "o, ngờ, ong", "ông": "ô, ngờ, ông", "ung": "u, ngờ, ung", "ưng": "ư, ngờ, ưng", "uông": "u, ô, ngờ, uông", "eng": "e, ngờ, eng", "iêng": "i, ê, ngờ, iêng", "uât": "u, ớ, tờ, uất", "uân": "u, ớ, nờ, uân", "uây": "u, ớ, i, uây", "oet": "o, e, tờ, oét", "uyu": "u, y, u, uyu" }; // Nếu vần nằm trong nhóm phức tạp, trả về ngay cách đọc chuẩn if (danhSachVanPhucTap[word]) { return danhSachVanPhucTap[word]; } // 2. Tự động dịch các vần 2 chữ cái (am, an, at, et, ot, uc...) sang từ thuần Việt để ép loa đọc chuẩn let components = []; for (let i = 0; i < word.length; i++) { let char = word[i]; // Ép trình duyệt đọc đúng tên âm vị tiếng Việt bằng các từ có dấu thanh nhẹ if (char === 'n') components.push("nờ"); else if (char === 'm') components.push("mờ"); else if (char === 't') components.push("tờ"); else if (char === 'c') components.push("cờ"); else if (char === 'p') components.push("pờ"); else if (char === 'g') components.push("gờ"); else if (char === 'h') components.push("hờ"); else if (char === 'b') components.push("bờ"); else if (char === 'd') components.push("giờ"); else if (char === 'đ') components.push("đờ"); else if (char === 'l') components.push("lờ"); else if (char === 'r') components.push("rờ"); else if (char === 's') components.push("sờ"); else if (char === 'v') components.push("vờ"); else if (char === 'x') components.push("xờ"); // Xử lý phụ âm ghép tiếng Việt ở cuối vần else if (word.substr(i, 2) === 'ch') { components.push("chờ"); i++; } else if (word.substr(i, 2) === 'ng') { components.push("ngờ"); i++; } else if (word.substr(i, 2) === 'nh') { components.push("nhờ"); i++; } // Giữ nguyên các nguyên âm nhưng ép bóc tách độc lập else { components.push(char); } } // Nối các âm lại bằng dấu phẩy để ép máy đọc ngắt quãng (Ví dụ: "e, tờ, et") let guideText = components.join(", "); return `${guideText}, ${word}`; }
